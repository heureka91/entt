80 static void fill_vtable_entry(Ret (*&entry)(Any &, Args...)) noexcept {
81 if constexpr(std::is_invocable_r_v<Ret, decltype(Candidate), Args...>) {
82 entry = +[](Any &, Args... args) -> Ret {
